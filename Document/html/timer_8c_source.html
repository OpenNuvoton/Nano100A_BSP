87 {

93  timer->CTL = 0;
94 
95  if(u32Clk == 10000) // min delay is 100us if timer clock source is LIRC 10k
96  {
97  u32Usec = ((u32Usec + 99) / 100) * 100;
98  }
99  else // 10 usec every step
100  {
101  u32Usec = ((u32Usec + 9) / 10) * 10;
102  }
103 
104  if(u32Clk > 0xFFFFFF) // For Nano100, only needs to consider 32MHz at most
105  {
106  u32Prescale = 1;
107  u32Clk >>= 1;
108  }
109 
110  // u32Usec * u32Clk might overflow if using uint32_t
111  fCmpr = ((float)u32Usec * (float)u32Clk) / 1000000.0;
112

114  timer->PRECNT = u32Prescale;
115  timer->CTL = TIMER_CTL_TMR_EN_Msk; // one shot mode
116
